domain the domain identifier; for the site module it should be set to "public" route(server, ctx, next) could intercept the routing process here, or trigger the next router routeNode(server, ctx, resolved, next) could intercept a template rendering process after that the target node has been resolved routeCustom(server, ctx, next) the implementation of a specific config route injection var app = null; module.exports = { domain: 'public', init: function (application) { app = application; }, route: function (ctx, next) { next(); }, routeNode: function (server, ctx, resolved, next) { if (ctx.restricted) { console.log('restricted page; intercepted here, could redirect to a login page'); next(); } else { next(); } }, routeCustom: function (server, ctx, next) { next(); } }
