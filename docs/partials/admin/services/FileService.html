FileService contains the few wrapped functions for the file manager rest api

all methods request a path variable, which should be relative to the "files/"

FileService.get(path)

(function () {
    
    angular.module("jsnbt")
        .controller('AppController', function ($scope, FileService) {

            $scope.file = undefined;

            FileService.get('/folder/image.jpg').then(function (response) {
                $scope.file = response;
            }, function (error) {
                throw error;
            });

            $scope.files = undefined;

            FileService.get([
                '/folder/image1.jpg', 
                '/folder/image2.jpg', 
                '/folder2'
            ]).then(function (response) {
                $scope.files = response;
            }, function (error) {
                throw error;
            });

        });

})();
