PagedDataService will bring you data from a collection in a paged wrapper object, including a more() function that will bring the next page's data.

The functions in this service are most suited for the ctrlGrid control.

PagedDataService.get(fn, query, start, limit, selected)

(function () {
    
    angular.module("jsnbt")
        .controller('AppController', function ($scope, PagedDataService) {

            $scope.data = {};

            PagedDataService.get(dpd.data.get, { 
                list: 'list1'
            }, 0, 50, []).then(function (results) {
                $scope.data = results;
            }, function (error) {
                throw error;
            });

        });

})();
