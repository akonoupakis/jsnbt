ModalService provides a wrapper function for the bootstrap $modal.open method.

ModalService.open(options)

The options object must specify a controller and a template to guide the modal function on the targeted html injected.
Besides that, everyting on the options will come in the target controller's $scope. 

(function () {

    angular.module("jsnbt")
        .controller('AppController', function ($scope, ModalService) {

            $scope.result = '';

            ModalService.open(function(x) {
                x.title('the modal window title');
                x.controller('ErrorPromptController');
                x.template('tmpl/core/modals/namePrompt.html');
                x.scope({
                    valueA: 'value 1',
                    valueB: 'value 2'
                });
            }).then(function (result) {
                $scope.result = result;
            }, function (error) {
                throw error;
            });

        });

})();
