ModalService provides a wrapper function for the bootstrap $modal.open method

ModalService.open(options)

The options object must specify a controller and a template to guide the modal function on the targeted html injected.
Besides that, everyting on the options will come in the target controller's $scope 

(function () {

    angular.module("jsnbt")
        .controller('AppController', function ($scope, ModalService) {

            $scope.result = '';

            var options = {
                // a custom title for the modal window
                title: 'the modal window title',
                // the controller to be used
                controller: 'NamePromptController',
                // the template to be used
                template: 'tmpl/core/modals/namePrompt.html',
                // any extra option properties, will be available in the controller's scope
                valueA: 'value 1',
                valueB: 'value 2'
            };

            ModalService.open(options).then(function (result) {
                $scope.result = result;
            }, function (error) {
                throw error;
            });

        });

})();
