AuthService contains a few common authentication and authorization functions

AuthService.login(username, password)

(function () {
    
    angular.module("jsnbt")
        .controller('AppController', function ($scope, AuthService) {

            $scope.user = undefined;

            AuthService.login('username', 'password').then(function (response) {
                $scope.user = response;
            }, function (error) {
                throw error;
            });

        });

})();

AuthService.get()

(function () {

    angular.module("jsnbt")
        .controller('AppController', function ($scope, AuthService) {
            
            AuthService.get().then(function (response) {
                $scope.user = response;
            }, function (error) {
                throw error;
            });

        });

})();

AuthService.logout()

(function () {

    angular.module("jsnbt")
        .controller('AppController', function ($scope, AuthService) {

            AuthService.logout().catch(function (error) {
                throw error;
            });

        });

})();

AuthService.isInRole(user, role)

(function () {
    angular.module("jsnbt")

        .controller('AppController', function ($scope, AuthService) {

            $scope.isAdmin = false;

            AuthService.get().then(function (response) {
                $scope.isAdmin = AuthService.isInRole(response, 'admin');
            }, function (error) {
                throw error;
            });
            
        });

})();

AuthService.authorize(user, section)

(function () {

    angular.module("jsnbt")

            $scope.permission = false;

            }, function (error) {
                throw error;
            });

        });

})();
