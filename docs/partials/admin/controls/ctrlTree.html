ngModel object
the model to be bound to the control
this object must have the results set along with a more() function that when triggered returns the next results page
this sort of function is provided by the PagedDataService that will be covered in next chapters
ngDomain text
the target domain to retrieve nodes for
ngFn object
an object with functions to be used internally in the control

usage:

(function () {
    angular.module("jsnbt")
        .controller('AppController', function ($scope) {

            $scope.treeFn = {
                canOpen: function (item) {
                    return true;
                },
                open: function (item) {
                    console.log('open', item);
                },
                canCreate: function (item) {
                    return true;
                },
                create: function (item) {
                    console.log('create', item);
                },
                canEdit: function (item) {
                    return true;
                },
                edit: function (item) {
                    console.log('edit', item);
                },
                canDelete: function (item) {
                    return true;
                },
                delete: function (item) {
                    console.log('delete', item);
                }
            };
        });
})();
