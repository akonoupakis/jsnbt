a grid control

ngModel object
the model to be bound to the control
this object must have the results set along with a more() function that when triggered returns the next results page
this sort of function is provided by the PagedDataService that will be covered in next chapters
ngFn object
an object with functions to be used internally in the control

usage:

(function () {

    angular.module("jsnbt")
        .controller('AppController', function ($scope) {
            
            $scope.gridFn = {
                edit: function (item) {
                    console.log('edit', item);
                },
                delete: function (item) {
                    console.log('delete', item);
                }
            };

        });
})();
<ctrl-grid ng-model="data" ng-fn="gridFn">
    <ctrl-grid-header>
        <ctrl-grid-header-column>name</ctrl-grid-header-column>
        <ctrl-grid-header-column></ctrl-grid-header-column>
    </ctrl-grid-header>
    <ctrl-grid-footer>
        <ctrl-grid-infinite-scroll></ctrl-grid-infinite-scroll>
        <ctrl-grid-empty>no data</ctrl-grid-empty>
    </ctrl-grid-footer>
    <ctrl-grid-body>
        <ctrl-grid-column>
            {{ row.name }}
        </ctrl-grid-column>
        <ctrl-grid-buttons-column>
            <button class="btn btn-default glyphicon glyphicon-edit btn-sm" ng-click="fn.edit(row)" />
            <button class="btn btn-default glyphicon glyphicon-remove btn-sm" ng-click="fn.delete(row)" />
        </ctrl-grid-buttons-column>
    </ctrl-grid-body>
</ctrl-grid>
