Each module uses an index entry point (typically the package.json main js file reference) to inject its functionality and parameterization to the jsnbt engine.
The index file is placed in the "src/app/" folder as all server side code blocks.

The below are the specs needed in a module's index.js file: domain the module domain identifier init(application) runs on application start getConfig() returns the module's cofiguration as a json object getVersion() returns the module's current version getBower() returns the module's bower cofiguration as a json object route(server, ctx, next) could intercept the routing process here, or trigger the next router view.preparse(server, ctx, preparsingContext, next) could change the preparsingContext.model and the preparsingContext.tmpl before rendering view.postparse(server, ctx, postparsingContext, next) could change the postparsingContext.html upon render routePointer(server, ctx, next) specify a custom pointer routing process
the system invokes this method on a specific module if a pointer points to a node of the module routeApi(server, ctx, serviceName, fnName, fields, files, next) targets modules that expose a jsnbt api under /jsnbt-api/{domain}/{serviceName}/{fnName}/ for the module's domain var app = null; module.exports = { domain: 'public', public: true, init: function (application) { app = application; }, getConfig: function () { return require('../cfg/config.js'); }, getVersion: function () { return require('../../package.json').version; }, getBower: function () { return require('../src/web/bower.json'); }, route: function (ctx, next) { next(); }, routeApi: function (server, ctx, serviceName, fnName, fields, files, next) { next(); }, routePointer: function (server, ctx, next) { next(); }, view: { preparse: function (server, ctx, preparsingContext, next) { next(); }, postparse: function (server, ctx, postparsingContext, next) { next(); } } };
